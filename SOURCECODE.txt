int countdown()
{
    char a[10];
    for(int i=3;i>=0;i--)
        {
                if(kbhit())
                {
                return 0;
                }
                    delay(1000);
                    sprintf(a,"%d",i);
                    settextstyle(1, HORIZ_DIR, 3);
                    outtextxy(300,350,a);
                if(i==0)
                    exit(0);
        }
        return 0;
}

int fail()
{

    settextstyle(font, HORIZ_DIR, 2);
    outtextxy(x+100, y+20, "Press any key to restart");
    countdown();
    getch();
    return 0;
}
